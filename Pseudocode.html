1. Sum of Two Numbers

START
INPUT num1, num2
sum ← num1 + num2
OUTPUT sum
STOP

2. Even or Odd

START
INPUT num
IF num MOD 2 = 0 THEN
    OUTPUT "Even"
ELSE
    OUTPUT "Odd"
END IF
STOP

3. Largest of Three Numbers

START
INPUT a, b, c
IF a > b AND a > c THEN
    largest ← a
ELSE IF b > c THEN
    largest ← b
ELSE
    largest ← c
END IF
OUTPUT largest
STOP

4. Factorial

START
INPUT n
factorial ← 1
FOR i ← 1 TO n DO
    factorial ← factorial * i
END FOR
OUTPUT factorial
STOP

5. Linear Search

START
INPUT arr, target
FOR i ← 0 TO length(arr)-1 DO
    IF arr[i] = target THEN
        OUTPUT "Found at index", i
        STOP
    END IF
END FOR
OUTPUT "Not Found"
STOP

7. Reverse a String

START
INPUT text
reversed ← ""
FOR i ← length(text)-1 DOWNTO 0 DO
    reversed ← reversed + text[i]
END FOR
OUTPUT reversed
STOP

8. Count Vowels

START
INPUT text
count ← 0
FOR each char IN text DO
    IF char IN ['a','e','i','o','u','A','E','I','O','U'] THEN
        count ← count + 1
    END IF
END FOR
OUTPUT count
STOP

9. Minimum in Array

START
INPUT arr
min ← arr[0]
FOR each number IN arr DO
    IF number < min THEN
        min ← number
    END IF
END FOR
OUTPUT min
STOP

10. Check Prime Number

START
INPUT n
IF n < 2 THEN
    OUTPUT "Not Prime"
    STOP
END IF
FOR i ← 2 TO sqrt(n) DO
    IF n MOD i = 0 THEN
        OUTPUT "Not Prime"
        STOP
    END IF
END FOR
OUTPUT "Prime"
STOP
